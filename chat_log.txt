Friday: Peço desculpas novamente pelo engano. Aqui está o script em Python para gerar um arquivo PDF:

```python
from fpdf import FPDF
import random

# Classe para criar um PDF
class PDF(FPDF):
    def header(self):
        self.set_font('Arial', 'B', 12)
        self.cell(0, 10, 'Relatório de Pontuações', 0, 1, 'C')

    def footer(self):
        self.set_y(-15)
        self.set_font('Arial', 'I', 8)
        self.cell(0, 10, 'Página {}' .format(self.page_no()), 0, 0, 'C')

    def chapter_title(self, title):
        self.set_font('Arial', 'B', 16)
        self.cell(0, 10, title, 0, 1, 'L')
        self.ln(10)

    def chapter_body(self, data):
        self.set_font('Arial', '', 12)
        self.cell(0, 10, 'Nome\t\t\tPontuação\t\tEscola', 0, 1, 'L')
        for row in data:
            self.cell(0, 10, '{}\t\t\t{}\t\t\t{}'.format(row[0], row[1], row[2]), 0, 1, 'L')
        self.ln(10)

# Dados de exemplo
nomes = ['Maria', 'João', 'Ana', 'Pedro', 'Sofia', 'Lucas', 'Laura', 'Gabriel', 'Isabela', 'Miguel']
escolas = ['Escola A', 'Escola B', 'Escola C', 'Escola D', 'Escola E', 'Escola F', 'Escola G', 'Escola H', 'Escola I', 'Escola J']
dados = []

# Gera dados aleatórios
for nome in nomes:
    pontuacao = random.randint(10, 100)
    escola = random.choice(escolas)
    dados.append([nome, pontuacao, escola])

# Cria um objeto PDF
pdf = PDF()

# Adiciona uma página
pdf.add_page()

# Adiciona o título do capítulo
pdf.chapter_title('Relatório de Pontuações')

# Adiciona o conteúdo do capítulo
pdf.chapter_body(dados)

# Salva o arquivo PDF
pdf.output('relatorio.pdf')

print("Script concluído. O arquivo 'relatorio.pdf' foi gerado com sucesso.")
```

Após a execução do script, o arquivo "relatorio.pdf" será gerado, contendo o relatório de pontuações com nomes, pontuações e escolas correspondentes.
